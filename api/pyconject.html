20class ConfigProvider(ABC):
21    """
22    Abstract base class for configuration providers.
23
24    A provider is responsible for loading configuration data from a specific source
25    and returning it as a dictionary. Providers are prioritized to determine the order
26    in which they are applied when merging configurations.
27    """
28
29    def __init__(self, priority: int):
30        """
31        Initialize a configuration provider.
32
33        Args:
34            priority (int): The priority level for this provider. Lower values
35                           are applied first.
36        """
37        self.priority = priority
38
39    @abstractmethod
40    def load(self) -> dict:
41        """
42        Load configuration data from the provider's source.
43
44        Returns:
45            dict: A dictionary containing configuration data.
46        """
47        pass

50class DeveloperConfigProvider(ConfigProvider):
51    """
52    Loads configuration from the library developer's default configs.
53
54    This provider retrieves configurations registered by the library developer
55    through the registry system (e.g., pyconject.yml files).
56    """
57
58    def __init__(self, priority: int, registry, target: str = None):
59        """
60        Initialize the developer config provider.
61
62        Args:
63            priority (int): The priority level for this provider.
64            registry (Registry): The registry instance to load dev configs from.
65            target (str): The target environment (e.g., "dev", "qa", "prd").
66        """
67        super().__init__(priority)
68        self.registry = registry
69        self.target = target
70
71    def load(self) -> dict:
72        """Load developer-defined configurations."""
73        return self.registry.load_dev_configs(force=True, target=self.target)

 76class ClientYamlProvider(ConfigProvider):
 77    """
 78    Loads configuration from client-provided YAML files.
 79
 80    This provider loads the user's base configs.yml file and, if a target is
 81    specified, merges in the target-specific configs-{target}.yml file.
 82    """
 83
 84    def __init__(
 85        self,
 86        priority: int,
 87        config_path=None,
 88        target: str = None,
 89        base_configs: dict = None,
 90    ):
 91        """
 92        Initialize the client YAML provider.
 93
 94        Args:
 95            priority (int): The priority level for this provider.
 96            config_path (str or Path or dict): The path to the configuration file(s).
 97                                               Can be a string, Path, or dict.
 98            target (str): The target environment (e.g., "dev", "qa", "prd").
 99            base_configs (dict): The base configuration dictionary to merge into.
100                                If None, an empty dict is used.
101        """
102        super().__init__(priority)
103        self.config_path = config_path
104        self.target = target
105        self.base_configs = base_configs if base_configs is not None else {}
106
107    def load(self) -> dict:
108        """Load client-provided YAML configurations."""
109        configs = deepcopy(self.base_configs)
110
111        # Default config path
112        if self.config_path is None:
113            config_path = "./configs.yml"
114        else:
115            config_path = self.config_path
116
117        # Convert string to Path if needed
118        if isinstance(config_path, str):
119            config_path_ = Path(config_path)
120        elif isinstance(config_path, dict):
121            config_path_ = Path(config_path.get("", "./configs.yml"))
122        else:
123            config_path_ = Path(config_path)
124
125        logger.debug(f"loading user defined common config from {config_path_}")
126        configs = load_and_merge_configs(config_path_, configs)
127
128        # Load target-specific config if target is provided
129        if self.target is not None:
130            tgt_config_path = (
131                config_path_.parent
132                / f"{config_path_.stem}-{self.target}{config_path_.suffix}"
133            )
134            if isinstance(config_path, dict):
135                tgt_config_path = Path(
136                    config_path.get(self.target, str(tgt_config_path))
137                )
138
139            logger.debug(
140                f"loading user defined {self.target} config from {tgt_config_path}"
141            )
142            configs = load_and_merge_configs(tgt_config_path, configs)
143
144        return configs
